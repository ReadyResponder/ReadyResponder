def sign_me_in
  somebody = FactoryBot.create(:user)
  r = FactoryBot.create(:role, name: 'Editor')
  somebody.roles << r
  visit new_user_session_path
  fill_in 'user_email', :with => somebody.email
  fill_in 'user_password', :with => somebody.password
  click_on 'Sign in'
end
